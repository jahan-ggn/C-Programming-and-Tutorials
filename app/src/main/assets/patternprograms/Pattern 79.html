#include<stdio.h>
 #include<conio.h>
 void main() {
 
 int i,j,k;
 int p_height=5; // change it to increase or decrease 
              pyramid height
 
 clrscr();
 
 
  for(i=0; i<p_height; i++)
 {
  for(k=p_height-1; k>i; k--)
  { printf(" "); }
 
  for(j=0; j<=i; j++)
  {
  printf("%c ",i+65);  // 65-> ASCII value of 'A'
  }
 
  printf("\n");
 }
 
 
  for(i=p_height-2; i>=0; i--)
 {
  for(k=p_height-1; k>i; k--)
  { printf(" "); }
 
  for(j=i; j>=0; j--)
  {
  printf("%c ",i+65); // 65-> ASCII value of 'A'
  }
 
  printf("\n");
 }
 
 getch();
 }
