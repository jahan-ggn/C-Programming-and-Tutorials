#include<stdio.h>
 #include<conio.h>
 void main() {
 
   int size =5;
   
 /*
 change value of 'size' to increase or decrease the size of 
 pyramid
 */
 
  int s,i,j,space=size;
 
 for(s=1;s<=size;s++) {
 
 for(i=1; i<=s; i++) {
  printf("*"); }
 
 for(j=space; j>1; j--) {
  printf("  "); }
 
 for(i=s; i>=1; i--) {
  printf("*"); }
 
  printf("\n"); space--;
  }
 }
