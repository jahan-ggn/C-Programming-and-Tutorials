#include < stdio.h >

void main()
{
int amount, rate, time, ans;

printf("\nEnter Principal Amount : ");
scanf("%d", &amount);

printf("\nEnter Rate of Interest : ");
scanf("%d", &rate);

printf("\nEnter Period of Time : ");
scanf("%d", &time);

ans = (amount * rate * time)/100;
/*Simple interest formula*/
printf("\nSimple Interest : %d",ans);
}
