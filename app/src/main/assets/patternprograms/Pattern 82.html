#include<stdio.h>
 
 void main()
 {
 
 int n=5;  // size
 
 int i,j;
 
 for(i=1;i<=n;i++)
 {
   /* outer gap loop */
 
   for(j=n-1;j>=i;j--)
   {
     printf(" ");
   }
 
     printf("%d",i);
 
   /* inner gap loop */
 
   for(j=1;j<(i-1)*2;j++)
   {
     printf(" ");
   }
 
   if(i>1)
   {
     printf("%d",i);
   }
 
     printf("\n");
   }
 
 }
