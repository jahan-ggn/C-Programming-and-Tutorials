Lower Case

#include < stdio.h >
#include < string.h >

int main()
{
  char string[] = "Strlwr in C";

  printf("%s\n",strlwr(string));
/*strlwr() will print string in lower case*/

  return 0;
}

Upper Case

#include < stdio.h >
#include < string.h >

int main()
{
  char string[] = "strupr in c";

  printf("%s\n",strupr(string));
/*strupr() will print string in upper case*/

  return 0;
}
