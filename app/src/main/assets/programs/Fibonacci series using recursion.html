#include < stdio.h >

int Fibonacci(int);

main()
{
  int n, i = 0, c;

  printf("Enter the number of terms ");
  scanf("%d",&n);

  printf("First %d terms of Fibonacci series are :-\n", n);

  for ( c = 1 ; c < = n ; c++ )
  {
  printf("%d\n", Fibonacci(i));
  i++;
  }

  return 0;
}

int Fibonacci(int n)
{
  if ( n == 0 )
  return 0;
  else if ( n == 1 )
  return 1;
  else
  return ( Fibonacci(n-1) + Fibonacci(n-2) );
  /*adding Fibonacci of (n-1) & (n-2) by recursive calling it*/
}
